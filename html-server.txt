function toBase64(str) {

var bytes = [];

if (code < 128) {
bytes.push(code);
} else if (code < 2048) {
bytes.push((code >> 6) | 192, (code & 63) | 128);
} else {
bytes.push((code >> 12) | 224, ((code >> 6) & 63) | 128, (code & 63) | 128);
}
}
var out = '';

var b0 = bytes[j], b1 = bytes[j+1], b2 = bytes[j+2];

return out;
